043import com.google.gwt.core.client.JavaScriptObject;
044import com.google.gwt.core.client.JsArray;

049/**
050 * Twitter typeahead.js
051 *
052 * https://github.com/twitter/typeahead.js
053 *
054 * @author Florian Kremser <florian.kremser@sage.com>
055 */
056public class Typeahead<T> extends TextBox {
057    private Collection<? extends Dataset<T>> datasets;
058    private boolean highlight = false;
059    private boolean hint = true;
060    private int minLength = 1;
061
062    public Typeahead() {
063        List<T> empty = Collections.emptyList();
064        setDatasets(new CollectionDataset<T>(empty));
065    }
066
067    /**
068     * A typeahead is composed of one or more datasets. When an end-user
069     * modifies the value of a typeahead, each dataset will attempt to
070     * render suggestions for the new value.
071     *
072     * @param dataset a dataset for providing suggestions
073     */
074    public Typeahead(final Dataset<T> dataset) {
075        setDatasets(dataset);
076    }
077
078    public Typeahead(final Collection<? extends Dataset<T>> datasets) {
079        setDatasets(datasets);
080    }
081
082    public void setDatasets(final Dataset<T> dataset) {
083        this.datasets = Arrays.asList(dataset);
084    }
085
086    public void setDatasets(final Collection<? extends Dataset<T>> datasets) {
087        this.datasets = datasets;
088    }
089
090    @Override
091    public void setValue(final String value, final boolean fireEvents) {
092        setValueNative(getElement(), value);
093        super.setValue(value, fireEvents);
094    }
095
096    /**
097     * If {@code true}, when suggestions are rendered, pattern matches for the
098     * current query in text nodes will be wrapped in a {@code strong} element
099     * with the {@code tt-highlight} class. Defaults to {@code false}.
100     *
101     * @param highlight {@code true} to highlight pattern matches in suggestions
102     */
103    public void setHighlight(final boolean highlight) {
104        this.highlight = highlight;
105    }
106
107    /**
108     * If {@code false}, the typeahead will not show a hint. Defaults to {@code true}.
109     *
110     * @param hint {@code true} to show a hint
111     */
112    public void setHint(final boolean hint) {
113        this.hint = hint;
114    }
115
116    /**
117     * The minimum character length needed before suggestions start getting
118     * rendered. Defaults to 1.
119     *
120     * @param minLength minimum required input length for matching
121     */
122    public void setMinLength(final int minLength) {
123        this.minLength = minLength;
124    }
125
126    public HandlerRegistration addTypeaheadOpenedHandler(final TypeaheadOpenedHandler<T> handler) {
127        return addHandler(handler, TypeaheadOpenedEvent.getType());
128    }
129
130    public HandlerRegistration addTypeaheadClosedHandler(final TypeaheadClosedHandler<T> handler) {
131        return addHandler(handler, TypeaheadClosedEvent.getType());
132    }
133
134    public HandlerRegistration addTypeaheadCursorChangededHandler(final TypeaheadCursorChangedHandler<T> handler) {
135        return addHandler(handler, TypeaheadCursorChangedEvent.getType());
136    }
137
138    public HandlerRegistration addTypeaheadAutoCompletedHandler(final TypeaheadAutoCompletedHandler<T> handler) {
139        return addHandler(handler, TypeaheadAutoCompletedEvent.getType());
140    }
141
142    public HandlerRegistration addTypeaheadSelectedHandler(final TypeaheadSelectedHandler<T> handler) {
143        return addHandler(handler, TypeaheadSelectedEvent.getType());
144    }
145
146    /**
147     * Triggered when the dropdown menu of the typeahead is opened.
148     *
149     * @param event the event
150     */
151    private void onOpened(final Event event) {
152        TypeaheadOpenedEvent.fire(this, event);
153    }
154
155    /**
156     * Triggered when the dropdown menu of the typeahead is closed.
157     *
158     * @param event the event
159     */
160    private void onClosed(final Event event) {
161        TypeaheadClosedEvent.fire(this, event);
162    }
163
164    /**
165     * Triggered when the dropdown menu cursor is moved to a different suggestion.
166     *
167     * @param event the event
168     * @param suggestion the suggestion object
169     */
170    private void onCursorChanged(final Event event, final Suggestion<T> suggestion) {
171        TypeaheadCursorChangedEvent.fire(this, suggestion, event);
172    }
173
174    /**
175     * Triggered when the query is autocompleted. Autocompleted means the query was changed to the hint.
176     *
177     * @param event the event
178     * @param suggestion the suggestion object
179     */
180    private void onAutoCompleted(final Event event, final Suggestion<T> suggestion) {
181        TypeaheadAutoCompletedEvent.fire(this, suggestion, event);
182    }
183
184    /**
185     * Triggered when a suggestion from the dropdown menu is selected.
186     *
187     * @param event the event
188     * @param suggestion the suggestion object
189     */
190    private void onSelected(final Event event, final Suggestion<T> suggestion) {
191        TypeaheadSelectedEvent.fire(this, suggestion, event);
192    }
193
194    public void reconfigure() {
195        remove(getElement());
196        configure();
197    }
198
199    @Override
200    protected void onLoad() {
201        super.onLoad();
202        configure();
203    }
204
205    @Override
206    protected void onUnload() {
207        super.onUnload();
208        remove(getElement());
209    }
210
211    protected void configure() {
212        JsArray<JavaScriptObject> datasetJSOs = JsArray.createArray().cast();
213        for (Dataset<T> dataset : datasets) {
214            JavaScriptObject jso = toJSO(dataset);
215            datasetJSOs.push(jso);
216        }
217        configure(getElement(), highlight, hint, minLength, datasetJSOs);
218    }
